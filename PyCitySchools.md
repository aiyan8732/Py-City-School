```python
schools_csv = "raw_data/schools_complete.csv"
students_csv = "raw_data/students_complete.csv"
schools_df = pd.read_csv(schools_csv, encoding = "utf-8")
students_df = pd.read_csv(students_csv, encoding = "utf-8")
```


```python
#Total Schools
total_schools = len(schools_df["name"])
#Total Students
total_students = len(students_df["name"])
#Sum of the budgets
total_budget = schools_df["budget"].sum()
#average math score of all students
average_math_score = students_df["math_score"].mean()
#average reading score of all students
average_reading_score = students_df["reading_score"].mean()
# math score >= 70 is passing math, create a passing_math dataframe
passing_math_df = students_df.loc[students_df["math_score"] > 70]
passing_math = len(passing_math_df["Student ID"])/total_students*100
# reading score >= 70 is passing reading, create a passing_reading dataframe
passing_reading_df = students_df.loc[students_df["reading_score"] > 70]
passing_reading = len(passing_reading_df["Student ID"])/total_students*100
# average of math/reading passing reate
overall_passing_rate = (passing_math + passing_reading)/2
```


```python
#school summary overview table
district_summary = pd.DataFrame({
    "Total Schools":[total_schools],
    "Total Students":[total_students],
    "Total Budget":[total_budget],
    "Average Math Score":[average_math_score],
    "Average Reading Score":[average_reading_score],
    "% Passing Math":[passing_math],
    "% Passing Reading":[passing_reading],
    "Overall Passing Rate":[overall_passing_rate]
})
# arrange columns order
district_summary = district_summary[["Total Schools",
                                     "Total Students",
                                     "Total Budget",
                                     "Average Math Score",
                                     "Average Reading Score",
                                     "% Passing Math",
                                     "% Passing Reading",
                                     "Overall Passing Rate"
                                    ]]
# format students and budgets
district_summary["Total Students"] = district_summary["Total Students"].map("{:,}".format)
district_summary["Total Budget"] = district_summary["Total Budget"].map("${:,.2f}".format)
district_summary
```
